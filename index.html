5 __construct 
              
class Constructor {
  public $a;
  function __construct($a) {
    $this->a = $a / 100;
    echo $this->a;
  }
}
              
            
              
//es5
function ConstructorES5(a) {
  this.a = a / 100;
  console.log(this.a);
}
             
//es5 se
function ConstructorES5se(a) {
  this.a = a;
  //self-exec function
  (function() {
    this.a = this.a / 100;
  }.bind(this))();
  console.log(this.a);
}

and properties		 
              
class Chain {
  public $name;
  function methodOne() {
    echo "Hi one";
    $this->name = "methodOne";
    return $this;
  }
  
  function methodTwo() {
    echo "Hi two $this->name";
  }
}

$chain = new Chain();
$chain->methodOne()->methodTwo();
              
            
              
function ChainES5() {
  this.name = "";
  this.methodOne = function() {
    console.log("one");
    this.name = "methodOne";
    return this;
  }
  this.methodTwo = function() {
    console.log(`two ${this.name}`);
  }
}

const chainES5 = new ChainES5();
chainES5.methodOne().methodTwo();
