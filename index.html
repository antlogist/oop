// Object.freeze
function UserES5(name) {
  let userName = name;
  
  return Object.freeze ({
    getName: function() {
      return userName;
    },
    setName: function(name) {
      userName = name;
    }
  });
}

const userES5 = new UserES5("Anthony");
console.log(userES5.getName());

// Object.freeze & Symbol
function UserSymbol(name) {
  
  const symbol = Symbol();
  
  return Object.freeze ({
    [symbol]: name,
    getName: function() {
      return this[symbol];
    },
    setName: function(name) {
      this[symbol] = name;
    }
  });
}

const userSymbol = new UserSymbol("John");
console.log(userSymbol.getName());

// Parent.php
class ParentClass {
  public $parentName;
  public $parentSurename;
  function __construct($parentName, 
                       $parentSurename = "Parent Surename") {
    $this->parentName = $parentName;
    $this->parentSurename = $parentSurename;
  }
  function getParentName() {
    return "Get parent name: $this->parentName";
  }
}
            
// Child.php
class ChildClass extends ParentClass {
  public $childName;
  function __construct($childName, ParentClass $parentClass) {
    $this->childName = $childName;
    $this->parentName = $parentClass->parentName;
    $this->parentSurename = $parentClass->parentSurename;
  }
  
  function getChildName() {
    return "Get child name: $this->childName";
  }
}
              
            
              
// Inheritance.js
import ParentClass from "./Parent.js";
import ChildClass from "./Child.js";

const parent = new ParentClass("Parent Class");

const child = new ChildClass("Child Class", parent);

// Parent.js
export default function ParentClass(
  parentName,
  parentSurename = "Parent Surename"
) {
  this.parentName = parentName;
  this.parentSurename = parentSurename;
}

ParentClass.prototype.getParentName = function() {
  return `Get parent name: ${this.parentName}`;
}

// Child.js
import ParentClass from "./Parent.js";

export default function ChildClass(childName, parentClass) {
  ParentClass.call(this, parentClass.parentName, parentClass.parentSurename);
  this.childName = childName;
  
  this.getParentName = function() {
    return parentClass.getParentName();
  }
  
  this.getChildName = function() {
    return this.childName;
  }
}

// Parent.js
export default class ParentClass {
  constructor(parentName, parentSurename = "Parent Surename") {
    this.parentName = parentName;

// Child.js
import ParentClass from "./Parent.js";

export default class ChildClass extends ParentClass {
  constructor(childName, parentClass) {
    super(parentClass);
    this.childName = childName;
    this.parentName = parentClass.parentName;
    this.parentSurename = parentClass.parentSurename;
  }
  
  getParentName() {
    return super.getParentName();
  }

  getChildName() {
    console.log(`Get child name: ${this.childName}`);
  }
}
